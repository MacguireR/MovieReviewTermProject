Indexes to Create
-----------------
Run these in MySQL after loading data.sql:

-- Index 1: Speed up review lookups by movie (used on movie detail page)
CREATE INDEX idx_review_movieID ON Review(movieID);

-- Index 2: Speed up movie lookups by genre (used on genre page)
CREATE INDEX idx_moviegenre_genreID ON MovieGenre(genreID);


Affected Queries
----------------
Query 1 (affected by idx_review_movieID):
  SELECT u.username, r.rating, r.comment, r.date
  FROM Review r
  JOIN User u ON r.userID = u.userID
  WHERE r.movieID = ?
  Page: /movies/{id}

Query 2 (affected by idx_moviegenre_genreID):
  SELECT m.movieID, m.title, m.releaseDate
  FROM Movie m
  JOIN MovieGenre mg ON m.movieID = mg.movieID
  WHERE mg.genreID = ?
  Page: /genres/{id}
